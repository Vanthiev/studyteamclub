លេខ​នៅ​ក្នុង PHP មាន ២ ប្រភេទ

  1. ចំនួន​គត់ (int)
  2. ចំនួន​ទសភាគ (float) ។

គណនា

ប្រមាណវិធី

ធម្មតា កាលណា​តម្លៃ​មួយ​ជា​លេខ យើង​អាច​គណនា​វា​បាន ដោយ​ប្រើ​វិធី

  • បូក សញ្ញា +
  • ដក សញ្ញា -
  • គុណ សញ្ញា *
  • ចែក សញ្ញា / ។

នៅ​ក្នុង PHP មាន​វិធី​មួយ​ទៀត​ក្នុង​ការ​គណនា​លេខ ដែល​ត្រូវ​បាន​ហៅ​ថា "ម៉ូឌូឡូ" ដោយ​ប្រើ​សញ្ញា % ។ ម៉ូឌូឡូ គឺ​គណនា​ដូច​ទៅ​នឹង​ការ​ចែក​ដែរ ប៉ុន្តែ​ការ​ចែក យើង​យក​ចម្លើយ រីឯ​ម៉ូឌូ​ឡូ យើង​យក​សំណល់ ។

echo 2 + 3.5;   // 5.5
echo 3 - 5;     // -2
echo 3 * -1;    // -3
echo 5 / 3;     // 1.6666666666667
echo 5 % 3;     // 2 ព្រោះ 5/3 សំណល់ 2

ការ​សរសេរ​កាត់

ការ​ផ្លាស់​ប្ដូរ​តម្លៃ​នៃ​អញ្ញាត ដោយ​យក​តម្លៃ​ចាស់​មក​ប្រើ ក្នុង​លក្ខណៈ​ខ្លះ យើង​អាច​សរសេរ​កាត​ដូច​ខាងក្រោម ។

$number = 20;
// ប្ដូរ​តម្លៃ​ថ្មី ដោយ​យក​តម្លៃ​ចាស់​ទៅ​គណនា
$number = $number + 2;
// យើង​អាច​សរសេរ​កាត់
$number += 2;
// យើង​អាច​ប្រើ​ជាមួយ​សញ្ញា​ដទៃ​ទៀត
$number -= 2;   // ស្មើ​នឹង $number = $number - 2
$number *= 3;   // ស្មើ​នឹង $number = $number * 3
$number /= 5;   // ស្មើ​នឹង $number = $number / 5
$number %= 4;   // ស្មើ​នឹង $number = $number % 4

// សំរាប់​ការ​តម្កើន​តម្លៃ​មួយ យើង​អាច​សរសេរ
$number = $number + 1;
$number += 1;
$number++;
++$number;

// សំរាប់​ការ​បន្ថយ​តម្លៃ​មួយ យើង​អាច​សរសេរ
$number = $number - 1;
$number -= 1;
$number--;
--$number;

ភាព​ខុសគ្នា​រវាង ++ មុន និង​ក្រោយ​អញ្ញាត ។

$number = 20;
echo $number++;     // បង្ហាញ 20 ដដែល
echo $number;       // បង្ហាញ 21

$number = 20;
echo ++$number;     // បង្ហាញ 21
echo $number;       // បង្ហាញ 21

$number++ គឺ​យក​តម្លៃ​ដើម​របស់​វា​ទៅ​ប្រើ រួច​បាន​វា​តម្កើន​មួយ ។ រី​ឯ ++$number គឺ​យក​តម្លៃ​ដើម​របស់​វា​ទៅតម្កើន​មួយ រួច​បាន​យក​ទៅ​ប្រើ ។ ហើយ​ក៏​ដូច​គ្នា​ដែរ​សំរាប់​ការ​សរសេរ -- មុន និង​ក្រោយ​អញ្ញាត ។
